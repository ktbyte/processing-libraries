Name

stop()
Examples 
SoundFile soundFile;
boolean musicIsOn = false;

void setup() {
    soundFile = new SoundFile(this, "https://opengameart.org/sites/default/files/one_0.mp3");
    soundFile.load();
    soundFile.play();
}

void draw() {
    if (!musicIsOn && soundFile.isLoaded()) {
        soundFile.play();
        musicIsOn = true;
    }
}

void mousePressed() {
    soundFile.stop();
}
