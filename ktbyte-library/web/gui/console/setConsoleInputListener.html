Console console;
void setup() {
  size(600, 600);
  console = new Console(this, 100, 100, 200, 200);
  console.setTextSize(16);
  console.setConsoleInputListener(new ConsoleInputListener() {

    public void onConsoleInput(String variable, String value) {
      println(value);
    }
  });
}
