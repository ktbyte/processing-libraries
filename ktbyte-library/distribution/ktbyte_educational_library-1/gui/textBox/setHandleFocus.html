Name

setHandleFocus()
Examples 

TextBox textbox;
void setup() {
  size(600, 600);
  textbox = new TextBox(this, 50, 50, 180, 50);
  textbox.setHandleFocus(false);
}

void draw() {
    background(100,100,210);
}

void mousePressed() {
  if(mouseX > 50 && mouseX < 50 + 180 && mouseY > 50 && mouseY < 50 + 50) {
    // if the mouse press is inside the box
    textbox.setIsFocused(true);
  } else {
      textbox.setIsFocused(false);
  }
}
Description Sets whether the handling of the box focus will be manual or automatic. By default the focus handling is automatic.
Syntax
setHandleFocus(handleFocus)
Parameters
handleFocus boolean: true = automatic handling, false = manual handling
Returnsvoid
