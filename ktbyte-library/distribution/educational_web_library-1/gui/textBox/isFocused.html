Name

isFocused()
Examples 

TextBox textbox;
void setup() {
  size(600, 600);
  textbox = new TextBox(this, 50, 50, 180, 50);
  textbox.setBorderRoundings(10,0,10,0);
}

void draw() {
    background(100,100,210);
}

void keyPressed() {
    if (!textbox.isFocused()) {
      println("The textbox element is out of focus so you cannot enter text inside it.");
      println("Click inside it to make it focused!");  
  } else {
      println("Blinking cursor means the tetbox is focused.");
      println("Clicking outside the textbox will make the element to be out of focus.");
  }
